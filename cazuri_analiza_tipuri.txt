// Caz 01 - IF: conditia trebuie sa fie scalara.
// Eroare asteptata: the if condition must be a scalar value
struct S{
	int x;
	};

int main(){
	struct S s;
	if(s)return 1;
	return 0;
	}

// Caz 02 - WHILE: conditia trebuie sa fie scalara.
// Eroare asteptata: the while condition must be a scalar value
int main(){
	int v[3];
	while(v)return 1;
	return 0;
	}

// Caz 03 - RETURN: expresia returnata trebuie sa fie scalara.
// Eroare asteptata: the return value must be a scalar value
struct S{
	int x;
	};

int main(){
	struct S s;
	return s;
	}

// Caz 04 - RETURN: functiile void nu pot returna o valoare.
// Eroare asteptata: a void function cannot return a value
void main(){
	return 1;
	}

// Caz 05 - RETURN: functiile non-void trebuie sa returneze o valoare.
// Eroare asteptata: a non-void function must return a value
int main(){
	return;
	}

// Caz 06 - RETURN: tipul expresiei returnate trebuie sa fie convertibil la tipul functiei.
// Eroare asteptata: cannot convert the return expression type to the function return type
struct S{
	int x;
	};

struct S makeS(){
	return 1;
	}

// Caz 07 - ATRIBUIRE: destinatia trebuie sa fie left-value.
// Eroare asteptata: the assign destination must be a left-value
int main(){
	1=2;
	return 0;
	}

// Caz 08 - ATRIBUIRE: destinatia nu trebuie sa fie constanta.
// Eroare asteptata: the assign destination cannot be constant
int main(){
	int v[3];
	v=1;
	return 0;
	}

// Caz 09 - ATRIBUIRE: destinatia trebuie sa fie scalara.
// Eroare asteptata: the assign destination must be scalar
struct S{
	int x;
	};

int main(){
	struct S s;
	s=1;
	return 0;
	}

// Caz 10 - ATRIBUIRE: sursa trebuie sa fie scalara.
// Eroare asteptata: the assign source must be scalar
int main(){
	int x;
	int v[3];
	x=v;
	return 0;
	}

// Caz 24 - CAST: tipul la care se converteste nu poate fi structura.
// Eroare asteptata: cannot convert to a struct type
struct S{
	int x;
	};

int main(){
	return (struct S)1;
	}

// Caz 25 - CAST: o structura nu se poate converti.
// Eroare asteptata: cannot convert a struct
struct S{
	int x;
	};

int main(){
	struct S s;
	return (int)s;
	}

// Caz 26 - CAST: un array se poate converti doar la alt array.
// Eroare asteptata: an array can be converted only to another array
int main(){
	int v[3];
	return (int)v;
	}

// Caz 27 - CAST: un scalar se poate converti doar la alt scalar.
// Eroare asteptata: a scalar can be converted only to another scalar
int main(){
	int x;
	x=(int[3])1;
	return 0;
	}

// Caz 28 - Minus unar: operandul trebuie sa fie scalar.
// Eroare asteptata: unary - must have a scalar operand

	};

	struct S s;
	return -s;
	}

// Caz 29 - NOT unar: operandul trebuie sa fie scalar, iar rezultatul este int.
// Eroare asteptata: unary ! must have a scalar operand

	int x;
	};

int main(){
	struct S s;
	return !s;
	}

	}

// Caz 32 - Selectie camp: operatorul . se aplica doar structurilor.

// Caz 35 - Apel functie: doar functiile pot fi apelate.

// Caz 36 - Functie folosita fara apel: o functie poate fi doar apelata.

	return f;
	}

// Caz 37 - Apel functie: prea multe argumente fata de lista de parametri.

	return f(1,2);
	}

	}

// Caz 39 - Apel functie: tipul argumentului trebuie sa fie convertibil la tipul parametrului.
// Eroare asteptata: in call, cannot convert the argument type to the parameter type
struct S{
	int x;
	};

int f(struct S s){
	return 0;
	}

int main(){
	return f(1);
	}

// Caz 40 - Caz valid: combinatii scalare acceptate, rezultate int pentru ||, &&, ==, !=, <, <=, >, >= si !.
// Eroare asteptata: niciuna, codul ar trebui acceptat.
int f(int x,double y,char c){
	int r;
	double d;
	r=x||c;
	r=x&&c;
	r=x==c;
	r=x!=c;
	r=x<y;
	r=x<=y;
	r=y>x;
	r=y>=x;
	r=!x;
	d=-y;
	d=x+y*2.0;
	return r;
	}

int main(){
	return f(1,2.0,'a');
	}
